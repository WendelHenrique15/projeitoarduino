// C++ code

int counter;
int counter2;
int counter3;
void setup()

{
pinMode(12, OUTPUT);
pinMode(8, OUTPUT);
pinMode(4, OUTPUT);
}

void loop()
{
for (counter = 0; counter < 2; ++counter) {
digitalWrite(12, HIGH);
digitalWrite(8, LOW);
delay(7000); // Wait for 7000 millisecond(s)
digitalWrite(12, LOW);
digitalWrite(4, HIGH);
delay(5000); // Wait for 5000 millisecond(s)
digitalWrite(8, HIGH);
digitalWrite(4, LOW);
delay(1000); // Wait for 1000 millisecond(s)
}
for (counter2 = 0; counter2 < 5; ++counter2) {
digitalWrite(8, HIGH);
delay(2000); // Wait for 2000 millisecond(s)
digitalWrite(8, LOW);
delay(2000); // Wait for 2000 millisecond(s)
}
}
